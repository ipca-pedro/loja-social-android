// Ficheiro: routes/auth.js
const express = require('express');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const pool = require('../db');
const router = express.Router();

// Chave secreta (deve estar no .env!)
const JWT_SECRET = 'a-minha-chave-secreta-para-a-loja-social-12345';

// POST /api/auth/login
router.post('/login', async (req, res) => {
  try {
    const { email, password } = req.body;
    
    if (!email || !password) {
      return res.status(400).json({ success: false, message: 'Email e password são obrigatórios' });
    }

    const result = await pool.query('SELECT * FROM colaboradores WHERE email = $1', [email]);

    if (result.rows.length === 0) {
      return res.status(401).json({ success: false, message: 'Credenciais inválidas' });
    }

    const colaborador = result.rows[0];
    
    // Comparar a password com bcryptjs
    const validPassword = await bcrypt.compare(password, colaborador.password_hash);
    
    if (!validPassword) {
      return res.status(401).json({ success: false, message: 'Credenciais inválidas' });
    }

    // Gerar Token JWT
    const payload = { id: colaborador.id, email: colaborador.email, nome: colaborador.nome };
    const token = jwt.sign(payload, JWT_SECRET, { expiresIn: '8h' });

    res.json({
      success: true,
      message: 'Login realizado com sucesso',
      token: token
    });

  } catch (error) {
    console.error('Erro no login:', error);
    res.status(500).json({ success: false, message: 'Erro interno do servidor' });
  }
});

module.exports = router;
