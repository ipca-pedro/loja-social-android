const express = require('express');
const pool = require('../db');
const router = express.Router();

// GET /api/public/campanhas
router.get('/campanhas', async (req, res) => {
  try {
    const result = await pool.query('SELECT * FROM campanhas WHERE ativo = true');
    res.json({ success: true, data: result.rows });
    
  } catch (error) {
    console.error('Erro ao buscar campanhas pÃºblicas:', error);
    res.status(500).json({ success: false, message: 'Erro interno' });
  }
});

// GET /api/public/stock-summary
router.get('/stock-summary', async (req, res) => {
  try {
    // Exemplo de resumo: total de itens e categorias
    const totalQuery = await pool.query('SELECT SUM(quantidade_atual) as total FROM stock_items');
    const categoriasQuery = await pool.query('SELECT COUNT(*) as total FROM categorias');
    
    res.json({ 
        success: true, 
        data: {
            total_itens: totalQuery.rows[0].total || 0,
            total_categorias: categoriasQuery.rows[0].total || 0
        }
    });

  } catch (error) {
    console.error('Erro ao buscar resumo de stock:', error);
    res.status(500).json({ success: false, message: 'Erro interno' });
  }
});
